00103                     static index_type map(boundary_type t,IteratorType b,IteratorType e,std::locale const &l)
00104                     {
00105                         index_type result;
00106 
00107                         //
00108                         // Optimize for most common cases
00109                         //
00110                         // C++0x requires that string is continious in memory and all known
00111                         // string implementations
00112                         // do this because of c_str() support. 
00113                         //
00114 
00115                         if(linear_iterator_traits<char_type,IteratorType>::is_linear && b!=e)
00116                         {
00117                             char_type const *begin = &*b;
00118                             char_type const *end = begin + (e-b);
00119                             index_type tmp=std::use_facet<boundary_indexing<char_type> >(l).map(t,begin,end);
00120                             result.swap(tmp);
00121                         }
00122                         else {
00123                             std::basic_string<char_type> str(b,e);
00124                             index_type tmp = std::use_facet<boundary_indexing<char_type> >(l).map(t,str.c_str(),str.c_str()+str.size());
00125                             result.swap(tmp);
00126                         }
00127                         return result;
00128                     }
